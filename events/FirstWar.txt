			any_owned_province = { base_manpower = 2 }
		}
	}
	
	mean_time_to_happen = {
		months = 6
	}
	
	option = {
		name = firstwar.3.option.a #Horror
		
		random_owned_province = {
			limit = { 
				OR = {
					base_tax = 2
					base_production = 2
					base_manpower = 2
				}
				
				OR = {
					area = area_elwynn_forest
					area = area_duskwood
					area = area_westfall
				}
			}
		
			if = {
				limit = { base_tax = 2 }
				add_base_tax = -1
			}

				limit = { base_production = 2 }
				add_base_production = -1
			}
			
			if = {
				limit = { base_manpower = 2 }
				add_base_manpower = -1
			}
		}
		
		hidden_effect = {
			random_country = {
				limit = { 
					culture_group = culture_group_orc

					is_at_war = yes
				}
